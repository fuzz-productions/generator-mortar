var React = require('react/addons');
var Fz    = FuzzyReaction.require('components', 'Form');

var DropdownSelectPage = React.createClass({
	mixins: [ResourceComponentMixin],

	getInitialState: function () {
		return {
			workingResource: {} FormStore will bind here
		}
	},

	dropdownOptions: ['Winter', 'Spring', 'Summer', 'Fall'],

	render: function() {
		return (
			<Fz.Form key="myFancyForm"
				formKey={this.formKey}
				bindResource={this.state.workingResource}>

				<Fz.Form.DropdownSelect fieldKey='season'
					options={this.dropdownOptions}
					multiple={false} placeholder='Select a Season'
					required={true} />

			</Fz.Form>
		);

	}
});

module.exports = DropdownSelectPage;
